class Node:
    def __init__(self,x):
        self.data=x
        self.next=None
class LinkedList:
    def __init__(self):
        self.head=self.tail=None
    def create(self,x):
        n=Node(x)
        if self.head==None:
            self.head=self.tail=n
        else:
            self.tail.next=n
            self.tail=n
    def show(self,x):
        
        temp=self.head
        while temp!=None:
            if temp.data==x:
                print(temp.data,"is present in the list ")
                break
            temp=temp.next
        else:
            print(x,"not present in the list")
